其他方式 ​

函数式弹框打开 ​

javascript
<script lang="ts">
  import { defineComponent, getCurrentInstance } from 'vue';
  import { openBaseGanttEditor } from '@crami/project';

  export default defineComponent({
    setup() {
      const instance = getCurrentInstance();
      /**
        * 打开甘特图编辑器
        */
      const openEditor = () => {
        openBaseGanttEditor(instance?.appContext, {
          projectId,
          calendarsApi,
          configApi,
          dataSourceApi,
          menusApi,
          resourcesApi,
          saveDataApi,
        });
      };

      return {
        openEditor,
      };
    },
  });
</script>
