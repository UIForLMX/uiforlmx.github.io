配置请求参数 ​

javascript
<template>
  <div style="height: 100%">
    <HBBaseGanttEditor
    :calendars-api="calendarsApi"
    :config-api="configApi"
    :data-source-api="dataSourceApi"
    :menus-api="menusApi"
    :resources-api="resourcesApi"
    :save-data-api="saveDataApi"
    :project-id="projectId"
    ></HBBaseGanttEditor>
  </div>
</template>

<script lang="ts">
  import { defineComponent } from 'vue';

  export default defineComponent({
    setup() {
      const projectId = 'PLM_PM_PROJECT_d9b82a9844e94d0b980e461a43007a87';
    
      const commonParams = {
        lang: 'zhs',
        connect: 'MYSQL',
        companyId: 'P_SYS_COMPANY_155f6a331e804b69a6fc47e6273699fc',
        userDept: 'P_SYS_DEPT_dz90005725;P_SYS_DEPT_dz90005725;',
        userGroup: 'P_SYS_COMPANY_155f6a331e804b69a6fc47e6273699fc',
        projectCode: 'PLM_PROJECT',
        userId: 'P_SYS_USER_99abfffd40da4d6eb66dc692d3f815fc',
        timeZone: '8',
      }

      const calendarsApi = {
        url: '/wbs/v1/data/getProjectCal',
        method: 'post',
        params: {
          ...commonParams,
        },
      };

      const configApi = {
        url: '/wbs/v1/data/getWbsEditorColumn',
        method: 'post',
        params: {
         ...commonParams,
        },
      };

      const dataSourceApi = {
        url: '/wbs/v1/data/list',
        method: 'post',
        params: {
         ...commonParams,
        },
      };

      const menusApi = {
        url: '/studio-cache/page/getMetaInfo',
        method: 'post',
        params: {
         ...commonParams,
        },
      };

      const resourcesApi = {
        url: '/wbs/v1/data/getProjectResource',
        method: 'post',
        params: {
          ...commonParams,
        },
      };

      const saveDataApi = {
        url: '/wbs/v1/data/save',
        method: 'post',
        params: {
          ...commonParams,
        },
      };

      return {
        calendarsApi,
        configApi,
        dataSourceApi,
        menusApi,
        resourcesApi,
        saveDataApi,
        projectId,
      };
    },
  });
</script>

函数式弹框打开 ​

javascript
<script lang="ts">
  import { defineComponent, getCurrentInstance } from 'vue';
  import { openBaseGanttEditor } from '@crami/project';

  export default defineComponent({
    setup() {
      const instance = getCurrentInstance();
      /**
        * 打开甘特图编辑器
        */
      const openEditor = () => {
        openBaseGanttEditor(instance?.appContext, {
          projectId,
          calendarsApi,
          configApi,
          dataSourceApi,
          menusApi,
          resourcesApi,
          saveDataApi,
        });
      };

      return {
        openEditor,
      };
    },
  });
</script>
